41      private static String buildJson(String name, String url) throws JsonProcessingException {
42          BookmarkVO vo = new BookmarkVO();
43          vo.setName(name);
44          vo.setUrl(url);
45  
46          String json = new ObjectMapper()
47                          .setSerializationInclusion(JsonInclude.Include.NON_NULL)
48                          .writeValueAsString(vo);
49          L.debug("request: json={}", json);
50  
51          return json;
52      }

54      private static ResultActions perform(MockMvc mvc, RequestBuilder request) throws Exception {
55          ResultActions result = mvc.perform(request);
56  
57          MockHttpServletResponse response = result.andReturn().getResponse();
58          L.debug("response: status={}, body={}", response.getStatus(), response.getContentAsString());
59  
60          return result;
61      }
