170     public void create_URLが重複した場合はBookmarkDuplicateException() {
171         // 準備
172         String name1 = "テスト　サイト1";
173         String url1 = "https://example.com/test1";
174 
175         this.bookmarkService.create(name1, url1);
176 
177         String name2 = "テスト　サイト2";
178         String url2 = "https://example.com/test1";
179 
180         try {
181             // 実行
182             this.bookmarkService.create(name2, url2);
183 
184             fail();
185         } catch (BookmarkDuplicateException e) {
186             // 結果確認
187             assertThat(e.getMessage(), is("url=https://example.com/test1 is duplicated."));
188         }
189 
190         assertThat(this.bookmarkRepo.count(), is(1L));
191 
192         Bookmark b = this.bookmarkRepo.findAll().get(0);
193 
194         assertThat(b.getId().length(), is(36));
195         assertThat(b.getName(), is("テスト　サイト1"));
196         assertThat(b.getUrl(), is("https://example.com/test1"));
197         assertNotNull(b.getUpdated());
198     }
